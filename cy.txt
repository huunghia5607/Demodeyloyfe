Dán Nội dung sau vào deploy.yml: (Đảm bảo đã thêm các Secrets FTP vào Repo GitHub trước đó)

YAML

# Tên của quy trình CI/CD
name: Deploy TrangCaNhan WebFE via FTP

on:
  push:
    branches:
      - main # Kích hoạt khi có thay đổi trên nhánh main

jobs:
  web-deploy:
    name: Deploy to Webhost
    runs-on: ubuntu-latest

    steps:
    # 1. Tải code từ GitHub
    - name: Checkout Code
      uses: actions/checkout@v4

    # 2. Triển khai bằng FTP
    - name: FTP Deploy to Somee Host
      uses: SamKirkland/FTP-Deploy-Action@v4.3.0
      with:
        server: ${{ secrets.FTP_SERVER }}
        username: ${{ secrets.FTP_USERNAME }}
        password: ${{ secrets.FTP_PASSWORD }}

        # Thư mục gốc code trên máy cục bộ
        local-dir: ./ 

        # Thư mục đích chính xác trên host
        remote-dir: /www.Nghia2k4.somee.com/PhamHuuNghia/

git add .github/workflows/deploy.yml
git commit -m "ci: add GitHub Actions workflow for TrangCaNhan FTP deployment"


git push origin main
